# Interactions
match (n:Interaction) Return n LIMIT 20

# Incident_Activities
match (n:Incident_Activity) Return n LIMIT 20

# Changes
match (n:Change) Return n LIMIT 20

# All nodes connected to one Incident (not further connected)
match (n:Incident)<-[r]-(e)
WHERE ID(n)=2
Return *

# Same for interaction
match (n)-[*0..2 ]-(e)
 WHERE ID(n)=3
 Return *
